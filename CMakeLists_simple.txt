# Simple CMakeLists.txt for WS2812 test only

cmake_minimum_required(VERSION 3.13)

set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

# Initialise pico_sdk from installed location
set(PICO_SDK_PATH "/Users/gx1000/code/pico-sdk")

set(PICO_BOARD pico CACHE STRING "Board type")

# Pull in Raspberry Pi Pico SDK (must be before project)
include(pico_sdk_import.cmake)

if (PICO_SDK_VERSION_STRING VERSION_LESS "1.4.0")
  message(FATAL_ERROR "Raspberry Pi Pico SDK version 1.4.0 (or later) required. Your version is ${PICO_SDK_VERSION_STRING}")
endif()

project(simple_ws2812_test C CXX ASM)

# Initialise the Raspberry Pi Pico SDK
pico_sdk_init()

# Generate WS2812 PIO header
pico_generate_pio_header(simple_ws2812_test ${CMAKE_CURRENT_LIST_DIR}/ws2812.pio OUTPUT_DIR ${CMAKE_CURRENT_LIST_DIR}/generated)

# Add executable
add_executable(simple_ws2812_test simple_ws2812_test.c )

pico_set_program_name(simple_ws2812_test "simple_ws2812_test")
pico_set_program_version(simple_ws2812_test "0.1")

# Enable stdio over UART
pico_enable_stdio_uart(simple_ws2812_test 1)
pico_enable_stdio_usb(simple_ws2812_test 0)

# Add the standard library to the build
target_link_libraries(simple_ws2812_test
        pico_stdlib
        hardware_pio)

# Add the standard include files to the build
target_include_directories(simple_ws2812_test PRIVATE
  ${CMAKE_CURRENT_LIST_DIR}
  ${CMAKE_CURRENT_LIST_DIR}/generated
)

pico_add_extra_outputs(simple_ws2812_test)
